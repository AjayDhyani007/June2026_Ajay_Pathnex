       #!/bin/bash

FILE="/etc/passwd"

while read line; do
  echo "Line: $line"
done < $FILE



pipeline {
    agent any
    environment {
        COMPONY_NAME = "DDLtech"
        SLACK_CHANNEL = "#devops"
    }
    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/Pathnex/sample-java-app.git'
            }
        }
        stage('Build') {
            steps {
                sh 'mvn clean package'
            }
        }
    }
    post {
        success {
            slackSend(channel: "$SLACK_CHANNEL", message: "SUCCESS: $COMPONY_NAME Build #${env.BUILD_NUMBER}")
        }
        failure {
            slackSend(channel: "$SLACK_CHANNEL", message: "FAILURE: $COMPONY_NAME Build #${env.BUILD_NUMBER}")
        }
    }
}

stages:
  - build

variables:
  COMPONY_NAME: "DDLtech"
  SLACK_WEBHOOK: "https://hooks.slack.com/services/XXXXX/YYYYY/ZZZZZ"

build:
  stage: build
  image: maven:3.8.1-jdk-17
  script:
    - git clone https://github.com/Pathnex/sample-java-app.git
    - cd sample-java-app
    - mvn clean package
    - curl -X POST -H 'Content-type: application/json' --data '{"text":"Build success for DDLtech"}' $SLACK_WEBHOOK
