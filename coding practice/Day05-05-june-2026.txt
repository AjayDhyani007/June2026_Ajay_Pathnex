




        # Day 05 - Handlers, User-Data, Resource Limits

        ## Ansible - Handler to restart Nginx

        -----
        -name: Manage Nginx
        hosts: all
        become: yes
task:
   - name: Update config file
    copy:
      src: index.html
      dest: /usr/share/nginx/html/index.html
     notify: Restart Nginx

  handlers:
   - name: Restart Ngonx
    service:
      name: nginx
      state: restarted
}
